# Types of Operators

# ORDER BY --> Used to sort the rows based on values in the column specified after ORDER BY.
#               (By default, it sorts in ascending order)
select * from employee order by employee_salary DESC;
# LIKE --> Used to search for a pattern in the value of a column for each row.
select * from employee where employee_name LIKE "%h%";
# BETWEEN --> Used to find values that lie within a specified range (inclusive of both ends).
select * from employee where employee_salary between 30000 and 50000;
# DISTINCT --> Shows only the unique (non-duplicate) rows.
select distinct * from employee;
select * from employee where employee_salary is NULL;
# AND --> Combines two conditions and returns the result only if both are true.
select * from employee where employee_salary > 50000 and employee_designation = 'Manager';
# OR --> Combines two conditions and returns the result if at least one of them is true.
select * from employee where employee_salary > 50000 or employee_designation = 'Manager';
# NOT --> Returns the result if the condition is false (negates the condition).
select * from employee where not employee_salary =  50000;
# IN --> Used to check whether a columnâ€™s value lies within a specified set of values.
select * from employee where employee_designation in ('Engineer','Manager','Technician');
# NULL --> Used to check whether a column has a NULL (missing) value.
insert into employee values(111,"Ram",NULL,'Manager',29);
# AS --> Used to give a temporary name (alias) to a column or table.
select employee_id as ID from employee;
# GROUP BY --> Groups rows that have the same value in the specified column(s).
select employee_designation, count(*) as number_of_employees from employee group by employee_designation;
# HAVING --> Used to filter the results of GROUP BY (similar to WHERE, but works on groups).
select employee_designation, count(*) as number_of_employees from employee group by employee_designation having count(*) > 2;
